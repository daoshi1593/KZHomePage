import * as vscode from 'vscode';
import * as fs from 'fs';
import * as path from 'path';

function readTasksJson() {
    // 
    const workspaceFolders = vscode.workspace.workspaceFolders;
    if (!workspaceFolders) {
        console.log('No workspace is open.');
        return;
    }
    const workspaceRoot = workspaceFolders[0].uri.fsPath;

    //  tasks.json 
    const tasksJsonPath = path.join(workspaceRoot, '.vscode', 'tasks.json');

    //  tasks.json 
    fs.readFile(tasksJsonPath, 'utf8', (err, data) => {
        if (err) {
            console.error('Error reading tasks.json:', err);
            return;
        }

        //  JSON 
        try {
            const tasksConfig = JSON.parse(data);
            console.log('Tasks configuration:', tasksConfig);
        } catch (parseErr) {
            console.error('Error parsing tasks.json:', parseErr);
        }
    });
}

function buildCommand(tasksConfig) {
    // 
    const workspaceFolder = '/path/to/your/project'; // 
    let command = tasksConfig.command;
    let args = tasksConfig.args.map(arg => arg.replace('${workspaceFolder}', workspaceFolder));

    // 
    const commandLine = `${command} ${args.join(' ')}`;
    console.log('Executing command:', commandLine);

    // 
    child_process.exec(commandLine, (error, stdout, stderr) => {
        if (error) {
            console.error(`Execution error: ${error}`);
            return;
        }
        console.log('stdout:', stdout);
        console.error('stderr:', stderr);
    });
}
