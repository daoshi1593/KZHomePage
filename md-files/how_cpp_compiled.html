常见之方式大调研

一些显而易见的事实

  • 任何高级语言都必须翻译成机器级指令 cpp无论何种方式都必须经过
  • 预处理器 ->.i
  • 编译器 ->.s
  • 汇编器 ->.o
  • 链接器 ->二进制可执行文件

factor 1 : ->gcc -o hello hello.c

此过程中指定的任务是将hello转换为hello可执行文件, 这个过程没有明确指定链接什么库,所以默认链接标准库libc

factor 2 : ->g++ -o ComplexProject src/main.cpp src/utils.cpp -I./include -L./lib -lboost_system -lboost_filesystem -lssl -lcrypto -std=c++17 -O2

项目结构示例 src/ main.cpp - 主程序入口 utils.cpp - 工具函数实现 include/ utils.h - 工具函数声明 lib/ 第三方库文件（例如，预编译的Boost和OpenSSL库） third_party/ 第三方源代码（如果需要从源代码编译）

单一命令行编译特点:

  • g++: 调用GNU C++编译器。
  • -o ComplexProject: 指定输出的可执行文件名为ComplexProject。
  • src/main.cpp src/utils.cpp: 指定要编译的源代码文件，这里有两个文件：main.cpp和utils.cpp。
  • -I./include: 指定头文件搜索路径为当前目录下的include文件夹。编译器会在这个路径下查找头文件。
  • -L./lib: 指定库文件搜索路径为当前目录下的lib文件夹。这告诉编译器在哪里查找链接时需要的库文件。
  • -lboost_system -lboost_filesystem -lssl -lcrypto: 链接到boost_system、 boost_filesystem、ssl和crypto这四个库。-l选项后面跟库名，编译器会在库文件搜索路径中查找这些库。
  • -std=c++17: 指定使用C++17标准进行编译。
  • -O2: 使用第二级优化，提高程序的运行效率，但编译时间可能会稍长

factor3: task.json

以初学之经历,vscode还能通过编写json配置文件,实现一键编译

import * as vscode from 'vscode';
import * as fs from 'fs';
import * as path from 'path';

function readTasksJson() {
    // 获取当前工作区的根目录路径
    const workspaceFolders = vscode.workspace.workspaceFolders;
    if (!workspaceFolders) {
        console.log('No workspace is open.');
        return;
    }
    const workspaceRoot = workspaceFolders[0].uri.fsPath;

    // 构建 tasks.json 文件的路径
    const tasksJsonPath = path.join(workspaceRoot, '.vscode', 'tasks.json');

    // 读取 tasks.json 文件
    fs.readFile(tasksJsonPath, 'utf8', (err, data) => {
        if (err) {
            console.error('Error reading tasks.json:', err);
            return;
        }

        // 解析 JSON 数据
        try {
            const tasksConfig = JSON.parse(data);
            console.log('Tasks configuration:', tasksConfig);
        } catch (parseErr) {
            console.error('Error parsing tasks.json:', parseErr);
        }
    });
}

// 假设这个函数在某个时刻被调用，例如扩展激活时
readTasksJson();

现在得到了json的配置数据,接下来干什么?


function buildCommand(tasksConfig) {
    // 替换变量占位符，这里仅作为示例，实际情况可能需要更复杂的处理
    const workspaceFolder = '/path/to/your/project'; // 假设的工作区目录
    let command = tasksConfig.command;
    let args = tasksConfig.args.map(arg => arg.replace('${workspaceFolder}', workspaceFolder));

    // 构建命令行字符串
    const commandLine = `${command} ${args.join(' ')}`;
    console.log('Executing command:', commandLine);

    // 执行命令行
    child_process.exec(commandLine, (error, stdout, stderr) => {
        if (error) {
            console.error(`Execution error: ${error}`);
            return;
        }
        console.log('stdout:', stdout);
        console.error('stderr:', stderr);
    });
}

factor4:cmake与现代构建,meta_programing

# 设置CMake的最小版本要求
cmake_minimum_required(VERSION 3.10)

# 设置项目名称
project(ComplexProject)

# 设置C++标准
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED True)

# 添加编译优化标志
set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -O2")

# 包含的头文件目录
include_directories(include)

# 链接的库文件目录
link_directories(lib)

# 指定生成目标
add_executable(ComplexProject src/main.cpp src/utils.cpp)

# 链接库文件
target_link_libraries(ComplexProject boost_system boost_filesystem ssl crypto)
